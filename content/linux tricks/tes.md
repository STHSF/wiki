# 查看系统基本信息

```bash
top 

uname -a

cat /etc/issue

cat /proc/version

cat /proc/cpuinfo

cat /proc/meminfo

free -m

hostname

lspci -tv

lsusb -tv

env  #

df -h

dmesg

uptime

chkconfig --list

whoami who w

rpm -qa

rpm -qf

rpm -ql <package-name>

rpm -qc <package-name>

rpm -qd <package-name>

rpm -qi <package-name>

ps -ef

ps -aux 
```
